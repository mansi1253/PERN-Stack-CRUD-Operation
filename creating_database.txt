open SQL shell(PG)and run these commands

1.
CREATE DATABASE product

2.
\c product

3.
CREATE TABLE category(
  id SERIAL PRIMARY KEY ,
  name VARCHAR(255) NOT NULL,
  isactive boolean default false,
  description TEXT NOT NULL
);

CREATE TABLE product(
  id SERIAL PRIMARY KEY,
  prodName VARCHAR(255) NOT NULL,
  ProdCat INT[],
  cost INT NOT NULL,
  description TEXT NOT NULL,
  isactive BOOLEAN DEFAULT false
);

3.
DELETE FROM product WHERE id=1;

insert into category (name, description, isactive) values ('cat-1 ', 'this is cat-1',true);

insert into product (prodName, ProdCat, cost, description, isactive) values ('prod-1 ', '{"1","2"}', 100, 'this is prod-1',true);
